# Evaluation function
RMSPE <- function(y,y_hat) {
  n <- length(y)
  s <- 0
  for (idx in 1:n) {
    if (y[idx] != 0) {
      s <- s+((y[idx]-y_hat[idx])/y[idx])^2}
    }
  
  return(sqrt(1/n*s))
}
